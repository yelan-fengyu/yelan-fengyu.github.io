import requests
import re
import parsel

模拟浏览器

headers = {
#用户代理 浏览器基本身份信息
‘User-Agent’: ‘Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/109.0.0.0 Safari/537.36 SLBrowser/9.0.0.10191 SLBChan/105’
}

请求链接

url = ‘https://b.faloo.com/1365505_1.html'# 发送请求
response = requests.get(url=url, headers=headers)
#把response.text转成可解析对象
selector = parsel.Selector(response.text)
#提取标题
title = selector.css(‘.nr_center .c_left .c_l_title h1::text’).get().split(‘ ‘)[-1]
#提取小说内容 把列表合并为字符串
content = ‘\n’.join(selector.css(‘.nr_center .c_left .noveContent p::text’).getall())
#保存数据，保存到txt
with open(title+ ‘.txt’,mode=’a’, encoding=’utf-8’) as f:
f.write(content)
print(title)

print(content)
